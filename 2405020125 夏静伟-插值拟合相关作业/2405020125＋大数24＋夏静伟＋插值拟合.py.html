170  # 3、读入海水图片haishui.jpg到变量Hai里，读入Horse.jpg到变量Hor里面，
171  # 然后完成下面任务：
172  # (1)取出Hor的红色分量图像给Hor_R，取出Hai的红色分量图像给Hai_R;
173  #
174  # (2)观察变量Hor_R和Hai_R的行数和列数，将Hai_R取出部分行和部分列到Hai_R1里，
175  # 使得Hai_R1和Hor_R的行数和列数一样，这里需要自己计算着取；
176  # (3)将Hai_R1和Hor_R相加，得到H, 并显示H图像，注意用参数cmap='gray';
177  # (4)取出H的行列数分别给m和n，并定义H1是和H一样大小的零矩阵，
178  # 然后进行双重for循环通过判断H矩阵的值来填补H1矩阵的值，
179  # 规则是：如果H某个i, j位置的值大于180的、H1对应位置的值填成255，
180  # 如果H某个i, j位置的值介于100到180之间、H1对应位置的值填成100，
181  # 对于H里小于100的值、H1里面对应位置一律填成0。
182  # 给出本大题所有代码和H、H1图像的显示图片。
183  import cv2
184  # (1) 读取图片并提取红色分量（OpenCV默认BGR格式，红色分量为索引2）
185  Hai = cv2.imread('haishui.jpg')
186  Hor = cv2.imread('horse.jpg')
187  Hai_R = Hai[:, :, 2]
188  Hor_R = Hor[:, :, 2]
189  # (2) 调整Hai_R尺寸与Hor_R一致
190  h_horse, w_horse = Hor_R.shape
191  h_sea, w_sea = Hai_R.shape
192  
193  # 计算切片起始位置（确保居中裁剪）
194  start_h = (h_sea - h_horse) // 2 if h_sea > h_horse else 0
195  start_w = (w_sea - w_horse) // 2 if w_sea > w_horse else 0
196  end_h = start_h + h_horse
197  end_w = start_w + w_horse
198  
199  Hai_R1 = Hai_R[start_h:end_h, start_w:end_w]
200  
201  # (3) 矩阵相加并显示灰度图像
202  H = Hor_R + Hai_R1  # 直接相加可能溢出，如需截断可使用cv2.add
203  plt.imshow(H, cmap='gray')
204  plt.title('H Image')
205  plt.axis('off')
206  plt.show()
207  
208  # (4) 双重循环生成H1矩阵
209  m, n = H.shape
210  H1 = np.zeros((m, n), dtype=np.uint8)  # 初始化零矩阵
211  
212  for i in range(m):
213      for j in range(n):
214          val = H[i, j]
215          if val > 180:
216              H1[i, j] = 255
217          elif 100 <= val <= 180:
218              H1[i, j] = 100
219          else:
220              H1[i, j] = 0
221  
222  # 可选：显示H1结果
223  plt.imshow(H1, cmap='gray')
224  plt.title('H1 Image')
225  plt.axis('off')
